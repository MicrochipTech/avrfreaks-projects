00162 W1::_Searcher(uint8_t command, Address &address, Token &token)
00163 {
00164 
00165     // Note: 1-based bit counting
00166     uint8_t  last_zero_path = 0;    // Last zero taken at a discrepancy
00167     uint8_t  current_bit    = 0;    // Current bit counter
00168     uint8_t  search_dir     = 0;    // 1 or 0 for next bit?
00169     uint8_t  bits;
00170 
00171 
00172     // Did we finish last time? Reset state for next call.
00173     if(token == 0xFF) {
00174         token = 0;
00175         return false;
00176     }
00177 
00178     // Always reset when beginning a new search
00179     if(!Reset()) {
00180         return false;
00181     }
00182 
00183     // Begin search
00184     last_zero_path = 0;
00185     WriteByte(command);
00186 
00187 
00188     // Scan down 64 bits of the ROM...
00189     while(++current_bit <= 64) {
00190 
00191         bits  = ReadBit();       // Next bit value
00192         bits |= ReadBit() << 1;  // Complement
00193 
00194         switch(bits) {
00195         case 0: // Discrepancy
00196             if(current_bit == token) {
00197 
00198                 // We took the 0 path, now take the 1 path
00199                 search_dir = 1;
00200 
00201             }else if(current_bit > token) {
00202 
00203                 // New discrepancy... take the 0 path
00204                 search_dir      = 0;
00205                 last_zero_path  = current_bit;
00206 
00207             }else{
00208 
00209                 // Old discrepancy... take the old path from the address
00210                 if(GetBit(address, current_bit-1) == 0) {
00211                     search_dir      = 0;
00212                     last_zero_path  = current_bit;
00213                 }else{
00214                     search_dir      = 1;
00215                 }
00216 
00217             }
00218             break;
00219 
00220         case 1: // Only a 1
00221             search_dir = 1;
00222             break;
00223 
00224         case 2: // Only a 0
00225             search_dir = 0;
00226             break;
00227 
00228         case 3: // No devices!
00229             // Fallthrough
00230         default:
00231             return false;
00232         }
00233         SetBit(address, current_bit-1, search_dir);
00234         WriteBit(search_dir);
00235     }
00236 
00237     token = last_zero_path;
00238 
00239     // Check if we're done (no more 0 paths for which we still need to search the 1's path)
00240     if(token == 0) {
00241         token = 0xFF;
00242     }
00243 
00244     return true;
00245 }
