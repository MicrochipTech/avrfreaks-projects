00057 double
00058 DSTherm::GetTemp(bool ferinheit)
00059 {
00060     uint16_t    temp;   // Raw
00061     double      dtemp;  // Converted
00062 
00063     // Bus setup
00064     if(!WaitForConversion()) return NAN;
00065     if(!_wire.Reset()) return NAN;
00066 
00067     // Select our device
00068     _wire.MatchROM(_address);
00069     _wire.WriteByte(DS_READ_SCRATCH);
00070 
00071     // Read only the first 2 bytes
00072     temp  = _wire.ReadByte();                   // LSB
00073     temp |= ((uint16_t)_wire.ReadByte() << 8);  // MSB
00074 
00075     // Stop transmission
00076     _wire.Reset();
00077 
00078     dtemp = temp;
00079     dtemp /= 16;
00080 
00081     // Do a conversion to F if necessary
00082     if(!ferinheit)
00083         return dtemp;
00084     return 1.8*dtemp + 32;
00085 }

00095     static const size_t MAX_COUNT = 15000;  // 750ms max for a conversion

00099         if(count++ > MAX_COUNT) {
