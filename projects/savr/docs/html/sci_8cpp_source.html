00133     // Set Baud Rate.
00134     uint16_t brate = (uint16_t)__GETBAUD(sysClock, baud);

00138     /* Enable Rx and Tx, and interrupt */

00141     /* Frame Format - 8 data, no parity */

00143     __CTRLC = __CTRLC_ENABLE | _BV(__CTRLC_UCSZ1) | _BV(__CTRLC_UCSZ0);// | _BV(UPM1) | _BV(UPM0);
