00069 static void
00070 MsDelay(uint32_t sysClock, double ms)
00071 {
00072     static const uint16_t maxDelayLoops = 65535;
00073 
00074     // How many 4-cycle loops to delay?
00075     uint32_t loops = ms * (0.001 / (4.0 / sysClock));
00076 
00077     while(loops) {
00078         uint16_t delay = MIN(loops, maxDelayLoops);
00079         loops -= delay;
00080         _delay_loop_2(delay);
00081     }
00082 }
