00043 void
00044 TWI::Init(uint32_t sysClock, uint32_t outputFreq)
00045 {
00046 
00047     // SCL Freq = CPU Freq / ( 16 + 2(TWBR)(PrescalerValue) ) 
00048     // TWBR =  ((CPU Freq / SCL Freq) - 16) / 2 / (PrescalerValue)
00049     TWBR = ((sysClock/outputFreq) - 16) / 2;
00050 
00051     // Enable interrupt, enable the TWI
00052     TWCR = _BV(TWINT) | _BV(TWEN);

00161 uint8_t
00162 TWI::Address(uint8_t address, uint8_t rw)
00163 {
00164     // Create start condition
00165     uint8_t state = 0;
00166     TWCR = _BV(TWINT) | _BV(TWSTA) | _BV(TWEN);
00167 
00168     TWI::Wait();
00169     state = TWI::State();
00170     if(state != TW_START && state != TW_REP_START) {
00171         return 1;
00172     }
00173 
00174     // Send address | RW (has waits...)
00175     TWI::Send((address<<1) | (!!rw));
00176 
00177     state = TWI::State();
00178     if(state != TW_MR_SLA_ACK && state != TW_MT_SLA_ACK) {
00179         return 1;
00180     }
00181     
00182     return 0;
00183 }
