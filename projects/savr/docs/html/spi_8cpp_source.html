00134 void
00135 SPI::Init(uint32_t sysClock, uint32_t spiFreq)
00136 {
00137     uint8_t divExp = 0;
00138 
00139     /* Master mode: MISO is Input; MOSI, and SCK is output */
00140     SPI_DDR &= ~(_BV(SPI_MISO));
00141     SPI_DDR |= _BV(SPI_MOSI) | _BV(SPI_SCK);
00142 
00143     /* Setup MISO with a pull-up resistor */
00144     SPI_PORT |= _BV(SPI_MISO);
00145 
00146     // Round down divider and find 2^x (highest bit)
00147     divExp = highestBit(sysClock/spiFreq);
00148 
00149     // Bounds. divExp to be subtracted by one in a few lines...
00150     // Sooo, our bounds are [1, FREQ_CFG_SIZE], not the normal [0, FREQ_CFG_SIZE-1] (both inclusive)
00151     if(divExp == 0) divExp = 1;
00152     if(divExp > FREQ_CFG_SIZE) divExp = FREQ_CFG_SIZE;
00153     divExp--;
00154 
00155     /* Setup SPCR
00156      * SPI enabled, master mode, fck/2, MSB first
00157      * Mode 0
00158      */
00159     SPCR |= _BV(SPE) | _BV(MSTR) | pgm_read_byte(&regFreqCfg[divExp].spcr);
00160     SPSR |= pgm_read_byte(&regFreqCfg[divExp].spsr);
00161 }
