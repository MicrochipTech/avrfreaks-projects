5 REM nRF24L01 TX Test on Mega2560
10 C:=0#INIT PAYLOAD COUNTER
15 RFI 1 0 1# POWER UP IN TX MODE, CHANNEL 0, 2 MBPS
20 RFB 0 4# 4 BYTES OF DATA RX0
25 RFF# FLUSH RX AND TX BUFFERS
30 PRINT "---------~"# PERFORATION
35 PRINT "PAYLOAD " ; PRI C# PRINT A HEADER
40 GOSUB 50 ; C:= C + 1# CALL TRANSMIT ROUTINE AND INCREMENT PAYLOAD COUNTER
45 SLP 7 ; GOTO 30# SLEEP 1 SECOND AND LOOP
50 M:=PEEK VPG@RTC3# FETCH 4 RTC BYTES 
55 N:=PEEK VPG@RTC2
60 O:=PEEK VPG@RTC1
65 P:=PEEK VPG@RTC0
70 PRI "D:" ; PRX M# SHOW THE TIMESTAMP TO USER
75 PRI "D:" ; PRX N
80 PRI "D:" ; PRX O
85 PRI "D:" ; PRX P
90 RFT M N O P# SEND RTC DATA
95 RET# RETURN TO CALLER
