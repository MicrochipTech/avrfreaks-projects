Appendix, the Files

C:\HOME\CC\lccwin32.bat

PATH=C:\LCC\BIN;C:\HOME\CC\LCCWBAT;C:\EXLCCW;%OPATH%
SET INCLUDE=C:\LCC\INCLUDE
MD C:\EXLCCW
set OE=.OBJ
set OD=LCCWIN32

C:\HOME\CC\LCCWBAT\ccs.bat

LCC %1.CPP
md %OD%
MOVE %1.OBJ %OD%

C:\HOME\CC\LCCWBAT\lnk.bat

LCCLNK %OD%\%1.OBJ
move  %OD%\%1.EXE \EXLCCW
