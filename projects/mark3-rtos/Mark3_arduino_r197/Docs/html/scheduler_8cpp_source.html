63 //---------------------------------------------------------------------------
64 void Scheduler::Schedule()
65 {
66  K_UCHAR ucPri = 0;
67 
68  // Figure out what priority level has ready tasks (8 priorities max)
69  ucPri = aucCLZ[m_ucPriFlag >> 4 ];
70  if (ucPri == 0xFF) { ucPri = aucCLZ[m_ucPriFlag & 0x0F]; }
71  else { ucPri += 4; }
72 
73  // Get the thread node at this priority.
74  g_pstNext = (Thread*)( m_aclPriorities[ucPri].GetHead() );
75  g_ucFlag = m_ucPriFlag;
76 
77  KERNEL_TRACE_1( STR_SCHEDULE_1, (K_USHORT)g_pstNext->GetID() );
78 }
