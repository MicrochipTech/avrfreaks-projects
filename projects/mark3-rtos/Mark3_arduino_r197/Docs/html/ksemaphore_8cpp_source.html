95 //---------------------------------------------------------------------------
96 bool Semaphore::Post()
97 {
98  KERNEL_TRACE_1( STR_SEMAPHORE_POST_1, (K_USHORT)g_pstCurrent->GetID() );
99 
100  bool bThreadWake = 0;
101  K_BOOL bBail = false;
102  // Increment the semaphore count - we can mess with threads so ensure this
103  // is in a critical section. We don't just disable the scheudler since
104  // we want to be able to do this from within an interrupt context as well.
105  CS_ENTER();
106 
107  // If nothing is waiting for the semaphore
108  if (m_clBlockList.GetHead() == NULL)
109  {
110  // Check so see if we've reached the maximum value in the semaphore
111  if (m_usValue < m_usMaxValue)
112  {
113  // Increment the count value
114  m_usValue++;
115  }
116  else
117  {
118  // Maximum value has been reached, bail out.
119  bBail = true;
120  }
121  }
122  else
123  {
124  // Otherwise, there are threads waiting for the semaphore to be
125  // posted, so wake the next one (highest priority goes first).
126  bThreadWake = WakeNext();
127  }
128 
129  CS_EXIT();
130 
131  // If we weren't able to increment the semaphore count, fail out.
132  if (bBail)
133  {
134  return false;
135  }
136 
137  // if bThreadWake was set, it means that a higher-priority thread was
138  // woken. Trigger a context switch to ensure that this thread gets
139  // to execute next.
140  if (bThreadWake)
141  {
142  Thread::Yield();
143  }
144  return true;
145 }

149 bool Semaphore::Pend_i( K_ULONG ulWaitTimeMS_ )
150 #else
151 void Semaphore::Pend_i( void )
152 #endif
153 {
154  KERNEL_TRACE_1( STR_SEMAPHORE_PEND_1, (K_USHORT)g_pstCurrent->GetID() );
155 
156 #if KERNEL_USE_TIMEOUTS
157  Timer clSemTimer;
158  bool bUseTimer = false;
159 #endif
160 
161  // Once again, messing with thread data - ensure
162  // we're doing all of these operations from within a thread-safe context.
163  CS_ENTER();
164 
165  // Check to see if we need to take any action based on the semaphore count
166  if (m_usValue != 0)
167  {
168  // The semaphore count is non-zero, we can just decrement the count
169  // and go along our merry way.
170  m_usValue--;
171  }
172  else
173  {
174  // The semaphore count is zero - we need to block the current thread
175  // and wait until the semaphore is posted from elsewhere.
176 #if KERNEL_USE_TIMEOUTS
177  if (ulWaitTimeMS_)
178  {
179  g_pstCurrent->SetExpired(false);
180  clSemTimer.Init();
181  clSemTimer.Start(0, ulWaitTimeMS_, TimedSemaphore_Callback, (void*)this);
182  bUseTimer = true;
183  }
184 #endif
185  Block(g_pstCurrent);
186 
187  // Switch Threads immediately
188  Thread::Yield();
189  }
190 
191  CS_EXIT();
192 
193 #if KERNEL_USE_TIMEOUTS
194  if (bUseTimer)
195  {
196  clSemTimer.Stop();
197  return (g_pstCurrent->GetExpired() == 0);
198  }
199  return true;
200 #endif
201 }
