57

67

70

89 #endif

113 //---------------------------------------------------------------------------
114 void TimerList::Process(void)
115 {
116 #if KERNEL_TIMERS_TICKLESS
117  K_ULONG ulNewExpiry;
118  K_ULONG ulOvertime;
119  bool bContinue;
120 #endif
121 
122  Timer *pclNode;
123  Timer *pclPrev;
124 
125 #if KERNEL_USE_QUANTUM
126  Quantum::SetInTimer();
127 #endif
128 #if KERNEL_TIMERS_TICKLESS
129  // Clear the timer and its expiry time - keep it running though
130  KernelTimer::ClearExpiry();
131  do
132  {
133 #endif
134  pclNode = static_cast<Timer*>(GetHead());
135  pclPrev = NULL;
136 
137 #if KERNEL_TIMERS_TICKLESS
138  bContinue = 0;
139  ulNewExpiry = MAX_TIMER_TICKS;
140 #endif
141 
142  // Subtract the elapsed time interval from each active timer.
143  while (pclNode)
144  {
145  // Active timers only...
146  if (pclNode->m_ucFlags & TIMERLIST_FLAG_ACTIVE)
147  {
148  // Did the timer expire?
149 #if KERNEL_TIMERS_TICKLESS
150  if (pclNode->m_ulTimeLeft <= m_ulNextWakeup)
151 #else
152  pclNode->m_ulTimeLeft--;
153  if (0 == pclNode->m_ulTimeLeft)
154 #endif
155  {
156  // Yes - set the "callback" flag - we'll execute the callbacks later
157  pclNode->m_ucFlags |= TIMERLIST_FLAG_CALLBACK;
158 
159  if (pclNode->m_ucFlags & TIMERLIST_FLAG_ONE_SHOT)
160  {
161  // If this was a one-shot timer, deactivate the timer.
162  pclNode->m_ucFlags |= TIMERLIST_FLAG_EXPIRED;
163  pclNode->m_ucFlags &= ~TIMERLIST_FLAG_ACTIVE;
164  }
165  else
166  {
167  // Reset the interval timer.
169  // I think we're good though...
170  pclNode->m_ulTimeLeft = pclNode->m_ulInterval;
171 
172 #if KERNEL_TIMERS_TICKLESS
173  // If the time remaining (plus the length of the tolerance interval)
174  // is less than the next expiry interval, set the next expiry interval.
175  if ((pclNode->m_ulTimeLeft + pclNode->m_ulTimerTolerance) < ulNewExpiry)
176  {
177  ulNewExpiry = pclNode->m_ulTimeLeft + pclNode->m_ulTimerTolerance;
178  }
179 #endif
180  }
181  }
182 #if KERNEL_TIMERS_TICKLESS
183  else
184  {
185  // Not expiring, but determine how K_LONG to run the next timer interval for.
186  pclNode->m_ulTimeLeft -= m_ulNextWakeup;
187  if (pclNode->m_ulTimeLeft < ulNewExpiry)
188  {
189  ulNewExpiry = pclNode->m_ulTimeLeft;
190  }
191  }
192 #endif
193  }
194  pclNode = static_cast<Timer*>(pclNode->GetNext());
195  }
196 
197  // Process the expired timers callbacks.
198  pclNode = static_cast<Timer*>(GetHead());
199  while (pclNode)
200  {
201  pclPrev = NULL;
202 
203  // If the timer expired, run the callbacks now.
204  if (pclNode->m_ucFlags & TIMERLIST_FLAG_CALLBACK)
205  {
206  // Run the callback. these callbacks must be very fast...
207  pclNode->m_pfCallback( pclNode->m_pclOwner, pclNode->m_pvData );
208  pclNode->m_ucFlags &= ~TIMERLIST_FLAG_CALLBACK;
209 
210  // If this was a one-shot timer, let's remove it.
211  if (pclNode->m_ucFlags & TIMERLIST_FLAG_ONE_SHOT)
212  {
213  pclPrev = pclNode;
214  }
215  }
216  pclNode = static_cast<Timer*>(pclNode->GetNext());
217 
218  // Remove one-shot-timers
219  if (pclPrev)
220  {
221  Remove(pclPrev);
222  }
223  }
224 
225 #if KERNEL_TIMERS_TICKLESS
226  // Check to see how much time has elapsed since the time we
227  // acknowledged the interrupt...
228  ulOvertime = KernelTimer::GetOvertime();
229 
230  if( ulOvertime >= ulNewExpiry ) {
231  m_ulNextWakeup = ulOvertime;
232  bContinue = 1;
233  }
234 
235  // If it's taken longer to go through this loop than would take us to
236  // the next expiry, re-run the timing loop
237 
238  } while (bContinue);
239 
240  // This timer elapsed, but there's nothing more to do...
241  // Turn the timer off.
242  if (ulNewExpiry >= MAX_TIMER_TICKS)
243  {
244  KernelTimer::Stop();
245  }
246  else
247  {
248  // Update the timer with the new "Next Wakeup" value, plus whatever
249  // overtime has accumulated since the last time we called this handler
250  m_ulNextWakeup = KernelTimer::SetExpiry(ulNewExpiry + ulOvertime);
251  }
252 #endif
253 #if KERNEL_USE_QUANTUM
254  Quantum::ClearInTimer();
255 #endif
256 }
