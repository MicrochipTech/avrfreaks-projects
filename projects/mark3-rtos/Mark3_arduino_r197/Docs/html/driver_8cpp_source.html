66 {
67  K_CHAR *szTmp1 = (K_CHAR*) szStr1_;
68  K_CHAR *szTmp2 = (K_CHAR*) szStr2_;
69 
70  while (*szTmp1 && *szTmp2)
71  {
72  if (*szTmp1++ != *szTmp2++)
73  {
74  return 0;
75  }
76  }
77 
78  // Both terminate at the same length
79  if (!(*szTmp1) && !(*szTmp2))
80  {
81  return 1;
82  }
83 
84  return 0;
