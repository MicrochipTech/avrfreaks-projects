37 void ThreadPort::InitStack(Thread *pclThread_)
38 {
39  // Initialize the stack for a Thread
40  K_USHORT usAddr;
41  K_UCHAR *pucStack;
42  K_USHORT i;
43 
44  // Get the address of the thread's entry function
45  usAddr = (K_USHORT)(pclThread_->m_pfEntryPoint);
46 
47  // Start by finding the bottom of the stack
48  pucStack = (K_UCHAR*)pclThread_->m_pwStackTop;
49 
50  // clear the stack, and initialize it to a known-default value (easier
51  // to debug when things go sour with stack corruption or overflow)
52  for (i = 0; i < pclThread_->m_usStackSize; i++)
53  {
54  pclThread_->m_pwStack[i] = 0xFF;
55  }
56 
57  // Our context starts with the entry function
58  PUSH_TO_STACK(pucStack, (K_UCHAR)(usAddr & 0x00FF));
59  PUSH_TO_STACK(pucStack, (K_UCHAR)((usAddr >> 8) & 0x00FF));
60 
61  // R0
62  PUSH_TO_STACK(pucStack, 0x00); // R0
63 
64  // Push status register and R1 (which is used as a constant zero)
65  PUSH_TO_STACK(pucStack, 0x80); // SR
66  PUSH_TO_STACK(pucStack, 0x00); // R1
67 
68  // Push other registers
69  for (i = 2; i <= 23; i++) //R2-R23
70  {
71  PUSH_TO_STACK(pucStack, i);
72  }
73 
74  // Assume that the argument is the only stack variable
75  PUSH_TO_STACK(pucStack, (K_UCHAR)(((K_USHORT)(pclThread_->m_pvArg)) & 0x00FF)); //R24
76  PUSH_TO_STACK(pucStack, (K_UCHAR)((((K_USHORT)(pclThread_->m_pvArg))>>8) & 0x00FF)); //R25
77 
78  // Push the rest of the registers in the context
79  for (i = 26; i <=31; i++)
80  {
81  PUSH_TO_STACK(pucStack, i);
82  }
83 
84  // Set the top o' the stack.
85  pclThread_->m_pwStackTop = (K_UCHAR*)pucStack;
86 
87  // That's it! the thread is ready to run now.
88 }

106  Thread_Switch(); // Set the next scheduled thread to the current thread
