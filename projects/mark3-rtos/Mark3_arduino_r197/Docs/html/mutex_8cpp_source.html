96 #if KERNEL_USE_TIMEOUTS
97 bool Mutex::Claim_i(K_ULONG ulWaitTimeMS_)
98 #else
99 void Mutex::Claim_i(void)
100 #endif
101 {
102  KERNEL_TRACE_1( STR_MUTEX_CLAIM_1, (K_USHORT)g_pstCurrent->GetID() );
103 
104 #if KERNEL_USE_TIMEOUTS
105  Timer clTimer;
106  bool bUseTimer = false;
107 #endif
108 
109  // Disable the scheduler while claiming the mutex - we're dealing with all
110  // sorts of private thread data, can't have a thread switch while messing
111  // with internal data structures.
112  Scheduler::SetScheduler(0);
113 
114  // Check to see if the mutex is claimed or not
115  if (m_bReady != 0)
116  {
117  // Mutex isn't claimed, claim it.
118  m_bReady = 0;
119  m_ucRecurse = 0;
120  m_ucMaxPri = g_pstCurrent->GetPriority();
121  m_pclOwner = g_pstCurrent;
122 
123  Scheduler::SetScheduler(1);
124 
125 #if KERNEL_USE_TIMEOUTS
126  return true;
127 #else
128  return;
129 #endif
130  }
131 
132  // If the mutex is already claimed, check to see if this is the owner thread,
133  // since we allow the mutex to be claimed recursively.
134  if (g_pstCurrent == m_pclOwner)
135  {
136  // Ensure that we haven't exceeded the maximum recursive-lock count
137  KERNEL_ASSERT( (m_ucRecurse < 255) );
138  m_ucRecurse++;
139 
140  // Increment the lock count and bail
141  Scheduler::SetScheduler(1);
142 #if KERNEL_USE_TIMEOUTS
143  return true;
144 #else
145  return;
146 #endif
147  }
148 
149  // The mutex is claimed already - we have to block now. Move the
150  // current thread to the list of threads waiting on the mutex.
151 #if KERNEL_USE_TIMEOUTS
152  if (ulWaitTimeMS_)
153  {
154  g_pstCurrent->SetExpired(false);
155  clTimer.Init();
156  clTimer.Start(0, ulWaitTimeMS_, (TimerCallback_t)TimedMutex_Calback, (void*)this);
157  bUseTimer = true;
158  }
159 #endif
160  Block(g_pstCurrent);
161 
162  // Check if priority inheritence is necessary. We do this in order
163  // to ensure that we don't end up with priority inversions in case
164  // multiple threads are waiting on the same resource.
165  if(m_ucMaxPri <= g_pstCurrent->GetPriority())
166  {
167  m_ucMaxPri = g_pstCurrent->GetPriority();
168 
169  Thread *pclTemp = static_cast<Thread*>(m_clBlockList.GetHead());
170  while(pclTemp)
171  {
172  pclTemp->InheritPriority(m_ucMaxPri);
173  if(pclTemp == static_cast<Thread*>(m_clBlockList.GetTail()) )
174  {
175  break;
176  }
177  pclTemp = static_cast<Thread*>(pclTemp->GetNext());
178  }
179  m_pclOwner->InheritPriority(m_ucMaxPri);
180  }
181 
182  // Done with thread data -reenable the scheduler
183  Scheduler::SetScheduler(1);
184 
185  // Switch threads if this thread acquired the mutex
186  Thread::Yield();
187 
188 #if KERNEL_USE_TIMEOUTS
189  if (bUseTimer)
190  {
191  clTimer.Stop();
192  return (g_pstCurrent->GetExpired() == 0);
193  }
194  return true;
195 #endif
196 }

216 //---------------------------------------------------------------------------
217 void Mutex::Release()
218 {
219  KERNEL_TRACE_1( STR_MUTEX_RELEASE_1, (K_USHORT)g_pstCurrent->GetID() );
220 
221  bool bSchedule = 0;
222 
223  // Disable the scheduler while we deal with internal data structures.
224  Scheduler::SetScheduler(0);
225 
226  // This thread had better be the one that owns the mutex currently...
227  KERNEL_ASSERT( (g_pstCurrent == m_pclOwner) );
228 
229  // If the owner had claimed the lock multiple times, decrease the lock
230  // count and return immediately.
231  if (m_ucRecurse)
232  {
233  m_ucRecurse--;
234  Scheduler::SetScheduler(1);
235  return;
236  }
237 
238  // Restore the thread's original priority
239  if (g_pstCurrent->GetCurPriority() != g_pstCurrent->GetPriority())
240  {
241  g_pstCurrent->SetPriority(g_pstCurrent->GetPriority());
242 
243  // In this case, we want to reschedule
244  bSchedule = 1;
245  }
246 
247  // No threads are waiting on this semaphore?
248  if (m_clBlockList.GetHead() == NULL)
249  {
250  // Re-initialize the mutex to its default values
251  m_bReady = 1;
252  m_ucMaxPri = 0;
253  m_pclOwner = NULL;
254  }
255  else
256  {
257  // Wake the highest priority Thread pending on the mutex
258  if(WakeNext())
259  {
260  // Switch threads if it's higher or equal priority than the current thread
261  bSchedule = 1;
262  }
263  }
264 
265  // Must enable the scheduler again in order to switch threads.
266  Scheduler::SetScheduler(1);
267  if(bSchedule)
268  {
269  // Switch threads if a higher-priority thread was woken
270  Thread::Yield();
271  }
272 }
