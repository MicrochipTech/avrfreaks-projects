91 //---------------------------------------------------------------------------
92 void Thread::Start(void)
93 {
94  // Remove the thread from the scheduler's "stopped" list, and add it
95  // to the scheduler's ready list at the proper priority.
96  KERNEL_TRACE_1( STR_THREAD_START_1, (K_USHORT)m_ucThreadID );
97 
98  CS_ENTER();
99  Scheduler::GetStopList()->Remove(this);
100  Scheduler::Add(this);
101  m_pclOwner = Scheduler::GetThreadList(m_ucPriority);
102  m_pclCurrent = m_pclOwner;
103 
104  if (Kernel::IsStarted())
105  {
106  if (m_ucPriority >= Scheduler::GetCurrentThread()->GetCurPriority())
107  {
108 #if KERNEL_USE_QUANTUM
109  // Deal with the thread Quantum
110  Quantum::RemoveThread();
111  Quantum::AddThread(this);
112 #endif
113  }
114  if (m_ucPriority > Scheduler::GetCurrentThread()->GetPriority())
115  {
116  Thread::Yield();
117  }
118  }
119  CS_EXIT();
120 }

123 void Thread::Stop()
124 {
125  bool bReschedule = 0;
126 
127  CS_ENTER();
128 
129  // If a thread is attempting to stop itself, ensure we call the scheduler
130  if (this == Scheduler::GetCurrentThread())
131  {
132  bReschedule = true;
133  }
134 
135  // Add this thread to the stop-list (removing it from active scheduling)
136  Scheduler::Remove(this);
137  m_pclOwner = Scheduler::GetStopList();
138  m_pclCurrent = m_pclOwner;
139  m_pclOwner->Add(this);
140 
141  CS_EXIT();
142 
143  if (bReschedule)
144  {
145  Thread::Yield();
146  }
147 }

196 //---------------------------------------------------------------------------
197 void Thread::Sleep(K_ULONG ulTimeMs_)
198 {
199  Semaphore clSemaphore;
200  Timer *pclTimer = g_pstCurrent->GetTimer();
201 
202  // Create a semaphore that this thread will block on
203  clSemaphore.Init(0, 1);
204 
205  // Create a one-shot timer that will call a callback that posts the
206  // semaphore, waking our thread.
207  pclTimer->Init();
208  pclTimer->SetIntervalMSeconds(ulTimeMs_);
209  pclTimer->SetCallback(ThreadSleepCallback);
210  pclTimer->SetData((void*)&clSemaphore);
211  pclTimer->SetFlags(TIMERLIST_FLAG_ONE_SHOT);
212 
213  // Add the new timer to the timer scheduler, and block the thread
214  TimerScheduler::Add(pclTimer);
215  clSemaphore.Pend();
216 }

263 void Thread::Yield()
264 {
265  CS_ENTER();
266 
267  // Run the scheduler
268  if (Scheduler::IsEnabled())
269  {
270  Scheduler::Schedule();
271 
272  // Only switch contexts if the new task is different than the old task
273  if (Scheduler::GetCurrentThread() != Scheduler::GetNextThread())
274  {
275 #if KERNEL_USE_QUANTUM

277  // and restart it for the new thread (if required).

280 #endif
281  Thread::ContextSwitchSWI();
282  }
283  }
284  else
285  {
286  Scheduler::QueueScheduler();
287  }

289  CS_EXIT();
