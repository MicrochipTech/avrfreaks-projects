164 void EventFlag::Set(K_USHORT usMask_)
165 {
166  Thread *pclPrev;
167  Thread *pclCurrent;
168  bool bReschedule = false;
169  K_USHORT usNewMask;
170 
171  CS_ENTER();
172 
173  // Walk through the whole block list, checking to see whether or not
174  // the current flag set now matches any/all of the masks and modes of
175  // the threads involved.
176 
177  m_usSetMask |= usMask_;
178  usNewMask = m_usSetMask;
179 
180  // Start at the head of the list, and iterate through until we hit the
181  // "head" element in the list again. Ensure that we handle the case where
182  // we remove the first or last elements in the list, or if there's only
183  // one element in the list.
184  pclCurrent = static_cast<Thread*>(m_clBlockList.GetHead());
185 
186  // Do nothing when there are no objects blocking.
187  if (pclCurrent)
188  {
189  // First loop - process every thread in the block-list and check to
190  // see whether or not the current flags match the event-flag conditions
191  // on the thread.
192  do
193  {
194  pclPrev = pclCurrent;
195  pclCurrent = static_cast<Thread*>(pclCurrent->GetNext());
196 
197  // Read the thread's event mask/mode
198  K_USHORT usThreadMask = pclPrev->GetEventFlagMask();
199  EventFlagOperation_t eThreadMode = pclPrev->GetEventFlagMode();
200 
201  // For the "any" mode - unblock the blocked threads if one or more bits
202  // in the thread's bitmask match the object's bitmask
203  if ((EVENT_FLAG_ANY == eThreadMode) || (EVENT_FLAG_ANY_CLEAR == eThreadMode))
204  {
205  if (usThreadMask & m_usSetMask)
206  {
207  pclPrev->SetEventFlagMode(EVENT_FLAG_PENDING_UNBLOCK);
208  pclPrev->SetEventFlagMask(m_usSetMask & usThreadMask);
209  bReschedule = true;
210 
211  // If the "clear" variant is set, then clear the bits in the mask
212  // that caused the thread to unblock.
213  if (EVENT_FLAG_ANY_CLEAR == eThreadMode)
214  {
215  usNewMask &=~ (usThreadMask & usMask_);
216  }
217  }
218  }
219  // For the "all" mode, every set bit in the thread's requested bitmask must
220  // match the object's flag mask.
221  else if ((EVENT_FLAG_ALL == eThreadMode) || (EVENT_FLAG_ALL_CLEAR == eThreadMode))
222  {
223  if ((usThreadMask & m_usSetMask) == usThreadMask)
224  {
225  pclPrev->SetEventFlagMode(EVENT_FLAG_PENDING_UNBLOCK);
226  pclPrev->SetEventFlagMask(usThreadMask);
227  bReschedule = true;
228 
229  // If the "clear" variant is set, then clear the bits in the mask
230  // that caused the thread to unblock.
231  if (EVENT_FLAG_ALL_CLEAR == eThreadMode)
232  {
233  usNewMask &=~ (usThreadMask & usMask_);
234  }
235  }
236  }
237  }
238  // To keep looping, ensure that there's something in the list, and
239  // that the next item isn't the head of the list.
240  while (pclPrev != m_clBlockList.GetTail());
241 
242  // Second loop - go through and unblock all of the threads that
243  // were tagged for unblocking.
244  pclCurrent = static_cast<Thread*>(m_clBlockList.GetHead());
245  bool bIsTail = false;
246  do
247  {
248  pclPrev = pclCurrent;
249  pclCurrent = static_cast<Thread*>(pclCurrent->GetNext());
250 
251  // Check to see if this is the condition to terminate the loop
252  if (pclPrev == m_clBlockList.GetTail())
253  {
254  bIsTail = true;
255  }
256 
257  // If the first pass indicated that this thread should be
258  // unblocked, then unblock the thread
259  if (pclPrev->GetEventFlagMode() == EVENT_FLAG_PENDING_UNBLOCK)
260  {
261  UnBlock(pclPrev);
262  }
263  }
264  while (!bIsTail);
265  }
266 
267  // If we awoke any threads, re-run the scheduler
268  if (bReschedule)
269  {
270  Thread::Yield();
271  }
272 
273  // Update the bitmask based on any "clear" operations performed along
274  // the way
275  m_usSetMask = usNewMask;
276 
277  // Restore interrupts - will potentially cause a context switch if a
278  // thread is unblocked.
279  CS_EXIT();
280 }
