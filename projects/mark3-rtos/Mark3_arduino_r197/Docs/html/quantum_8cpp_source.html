45 static void QuantumCallback(Thread *pclThread_, void *pvData_)
46 {
47  // Validate thread pointer, check that source/destination match (it's
48  // in its real priority list). Also check that this thread was part of
49  // the highest-running priority level.
50  if (pclThread_->GetPriority() >= Scheduler::GetCurrentThread()->GetPriority())
51  {
52  if (pclThread_->GetCurrent()->GetHead() != pclThread_->GetCurrent()->GetTail() )
53  {
54  bAddQuantumTimer = true;
55  pclThread_->GetCurrent()->PivotForward();
56  }
57  }
58 }
